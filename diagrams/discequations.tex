\newcommand{\scattblob}[1]{
    \raisebox{-0.45\height}{
        \begin{tikzpicture}[
                decoration={
                        markings,
                        mark=at position 0.6 with {\arrow[scale=1.2]{latex}}
                    }
            ]
            % Line 1
            \draw[postaction={decorate}] (-1,1) -- (-0.2,0.2);
            \draw[postaction={decorate}] (0.2,-0.2) -- (1,-1);
            
            % Line 2
            \draw[postaction={decorate}] (-1,-1) -- (-0.2,-0.2);
            \draw[postaction={decorate}] (0.2,0.2) -- (1,1);
            
            % Circle
            \draw[black, fill=gray!10] (0,0) circle (0.5) node[] {$#1$};
        \end{tikzpicture}
    }}

\newcommand{\unitarityblob}[2]{

            \draw[postaction={decorate}] (-2,-1) -- (-1.2,-0.2);
            \draw[postaction={decorate}] (1.2,0.2) -- (2,1);
            
            % Middle lines
            \draw[postaction={decorate}] (-1,0) to[out=65,in=115] (1,0);
            \draw[postaction={decorate}] (-1,0) to[out=-65,in=-115] (1,0);
            
            % Cross
            \draw[dashed] (-0.5,1) -- (0.5,-1);
            
            % Circle
            \draw[black, fill=gray!10] (-1,0) circle (0.5) node[] {$#1$};
            \draw[black, fill=gray!10] (1,0) circle (0.5) node[] {$#2$};
            
        \end{tikzpicture}
    }}
